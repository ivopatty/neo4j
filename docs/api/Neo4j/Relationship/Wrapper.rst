     def wrapper
       props.symbolize_keys!
       begin
         most_concrete_class = sorted_wrapper_classes
         wrapped_rel = most_concrete_class.constantize.new
       rescue NameError
         return self
       end
     
       wrapped_rel.init_on_load(self, self._start_node_id, self._end_node_id, self.rel_type)
       wrapped_rel
     end
